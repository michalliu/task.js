task.js

	var box1 = document.getElementById("box1");
	task()
	.run(function() {
		box1.style.left = 20 + "px";
	})
	.sleep(1000)
	.run(function () {
		box1.style.left = 80 + "px";
	})
	.sleep(2000)
	.run(function () {
		box1.style.left = 120 + "px";
	})
	.sleep(3000)
	.run(function () {
		box1.style.left = 160 + "px";
	})
	.done()
